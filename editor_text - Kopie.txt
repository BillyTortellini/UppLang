module Module
{
    Holder :: struct { value: int; }
    glob := 15;
    func :: () -> int {return 15;}
    get_val :: (holder: *Holder) -> int {return holder.value;}
    module Sub 
    {
        glob_x := false;
        sub_func :: (x: float) -> float {return x * 2;}
    }
}

main :: ()
{
    Module::glob = 20;
    Module::func();
    Module::Sub::glob_x = true;
    Module::Sub::sub_func(Module::glob);
    x : Module::Holder;
    x.value = 15;
    y := Module::get_val(*x);
}
