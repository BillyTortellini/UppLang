fac_iter :: (n:int) -> int
{
    prev := 1;
    curr := 1;
    i := 2;
    while (i < n)
    {
        old := curr;
        curr = prev + curr; 
        prev = old;
        i=i+1;
    }
    return curr;
}

fac :: (n:int) -> int
{
    if (n <= 2)  return 1; 
    return fac(n-1) + fac(n-2);
}
